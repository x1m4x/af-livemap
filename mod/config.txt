# Сканировать карту только когда в руке предмет "Rat Scanner".
#   true  — карта строится, лишь пока держишь Rat Scanner (по умолчанию).
#   false — сканирование работает всегда, предмет не нужен.
require_rat_scanner = true

# Подстрока имени предмета в руке для опознания сканера (регистр не важен).
# Меняй, только если детект не срабатывает: включи log_held_item = true,
# возьми Rat Scanner в руку, посмотри имя в UE4SS-консоли и впиши сюда.
scanner_item_match = ratscanner

# Диагностика: печатать имя предмета в руке в консоль UE4SS (раз в ~2 сек).
# Нужно только чтобы узнать точное scanner_item_match. Потом верни false.
log_held_item = false

# Полностью выключить лидар-сканирование (позиции игроков всё равно пишутся).
scan_enabled = true

# Лучей за тик (нагрузка). Больше — быстрее карта, но выше нагрузка. По умолч. 12.
scan_rays = 12

# Дальность луча в сантиметрах (100 см = 1 м). По умолчанию 6000 (60 м).
scan_range_cm = 6000
