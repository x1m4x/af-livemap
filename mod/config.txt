# Scan the map only while holding the "Rat Scanner" item.
# Сканировать карту только когда в руке предмет "Rat Scanner".
#   true  - map builds only while you hold the Rat Scanner (default).
#           карта строится, лишь пока держишь Rat Scanner (по умолчанию).
#   false - scanning always works, no item needed.
#           сканирование работает всегда, предмет не нужен.
require_rat_scanner = true

# Substring of the held item's class name to identify the scanner (case-insensitive).
# Подстрока имени предмета в руке для опознания сканера (регистр не важен).
# Change only if detection fails (see log_held_item below).
# Меняй только если детект не срабатывает (см. log_held_item ниже).
scanner_item_match = ratscanner

# Diagnostics: print the held item's name to the UE4SS console (~every 2s).
# Диагностика: печатать имя предмета в руке в консоль UE4SS (раз в ~2 сек).
# Use it to find the correct scanner_item_match, then set back to false.
# Нужно чтобы узнать точное scanner_item_match, потом верни false.
log_held_item = false

# Language of mod log messages: auto | en | ru
# Язык сообщений мода в логе: auto | en | ru
#   auto - follow the game's language / по языку игры (default / по умолчанию)
language = auto

# Auto-detect trader NPCs and put them on the map.
# Автоопределение NPC-торговцев и показ их на карте.
# Traveling traders are only recorded while the Employee Locator trinket is worn.
# Кочующие торговцы записываются, только если надет тринкет Employee Locator.
detect_traders = true

# Diagnostics: log every nearby NPC's class to the UE4SS console.
# Диагностика: печатать класс каждого NPC рядом в консоль UE4SS.
# Use it if a trader is not detected, then send the log. Потом верни false.
log_npcs = false

# Fully disable lidar scanning (player positions are still written).
# Полностью выключить лидар (позиции игроков всё равно пишутся).
scan_enabled = true

# Rays per tick (load). More = faster map, higher load. Default 12.
# Лучей за тик (нагрузка). Больше - быстрее карта, выше нагрузка. По умолч. 12.
scan_rays = 12

# Ray range in centimeters (100 cm = 1 m). Default 6000 (60 m).
# Дальность луча в сантиметрах (100 см = 1 м). По умолчанию 6000 (60 м).
scan_range_cm = 6000

# Where the rays go. 0 = the whole sphere around you (default): you map
# everything nearby without aiming, but the rays spread thin, so distant
# surfaces fill in slowly. A value in degrees fires them in a cone in the
# direction you are LOOKING instead - same ray count, much denser, so big
# rooms and long corridors map far faster. You then map what you look at.
# Suggested: 60-90. Values under 10 are raised to 10; 360 or more means 0.
#
# Куда летят лучи. 0 = вся сфера вокруг (по умолчанию): карта строится без
# прицеливания, но лучи размазаны, и дальние поверхности заполняются долго.
# Значение в градусах пускает их конусом ТУДА, КУДА ТЫ СМОТРИШЬ - лучей
# столько же, но плотность куда выше, поэтому большие залы и длинные коридоры
# сканируются намного быстрее. Тогда карта строится по направлению взгляда.
# Рекомендуется 60-90. Меньше 10 поднимается до 10; 360 и больше = 0.
scan_cone_deg = 0
